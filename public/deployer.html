#!/usr/bin/env node // TraiteurPro Auto-Deployer // Usage: node deploy.js YOUR_GITHUB_TOKEN const https = require('https'); const fs = require('fs'); const path = require('path'); const TOKEN = process.argv[2]; const REPO = 'terangaprestigegroup-star/traiteurpro'; if (!TOKEN) { console.log('Usage: node deploy.js ghp_VOTRE_TOKEN'); process.exit(1); } const FILES = [ { local: 'server.js', remote: 'server.js' }, { local: 'public/dashboard.html', remote: 'public/dashboard.html' }, { local: 'public/landing.html', remote: 'public/landing.html' }, { local: 'public/livreur.html', remote: 'public/livreur.html' }, { local: 'public/demo.html', remote: 'public/demo.html' }, { local: 'public/gif-maker.html', remote: 'public/gif-maker.html' }, ]; function ghRequest(method, path, body) { return new Promise((resolve, reject) => { const data = body ? JSON.stringify(body) : null; const req = https.request({ hostname: 'api.github.com', path: `/repos/${REPO}/contents/${path}`, method, headers: { 'Authorization': `token ${TOKEN}`, 'Accept': 'application/vnd.github.v3+json', 'User-Agent': 'TraiteurPro-Deployer', 'Content-Type': 'application/json', ...(data ? { 'Content-Length': Buffer.byteLength(data) } : {}) } }, res => { let body = ''; res.on('data', d => body += d); res.on('end', () => { try { resolve({ status: res.statusCode, data: JSON.parse(body) }); } catch(e) { resolve({ status: res.statusCode, data: body }); } }); }); req.on('error', reject); if (data) req.write(data); req.end(); }); } async function deployFile(localPath, remotePath) { const content = fs.readFileSync(localPath, 'utf8'); const encoded = Buffer.from(content, 'utf8').toString('base64'); // Get current SHA const existing = await ghRequest('GET', remotePath); const sha = existing.data?.sha; const result = await ghRequest('PUT', remotePath, { message: `🚀 Auto-deploy ${path.basename(remotePath)}`, content: encoded, branch: 'main', ...(sha ? { sha } : {}) }); if (result.status === 200 || result.status === 201) { console.log(`✅ ${remotePath}`); return true; } else { console.log(`❌ ${remotePath}: ${result.data?.message}`); return false; } } async function main() { console.log('🚀 TraiteurPro Auto-Deployer\n'); let ok = 0, err = 0; for (const file of FILES) { if (!fs.existsSync(file.local)) { console.log(`⚠️ ${file.local} — fichier introuvable, ignoré`); continue; } const success = await deployFile(file.local, file.remote); if (success) ok++; else err++; await new Promise(r => setTimeout(r, 500)); } console.log(`\n${ok} fichiers déployés, ${err} erreurs`); if (ok > 0) console.log('⏳ Railway redémarre automatiquement dans 2 minutes'); } main().catch(console.error);
